       <script>
           function date()
           {
             var a=new Date();
             var date=a.getDate();
             var month=a.getMonth()+1;
             var year=a.getFullYear();
             var curday=date+"/"+month+"/"+year;
             document.getElementById("day").innerHTML=curday;
           } 
        </script>
